# instalar git e GitHub CLI
sudo apt update
sudo apt install git gh -y

# configurar git
git config --global user.name "Kevi Pegoraro"
git config --global user.email "kevipegoraro@hotmail.com"

# logar no GitHub
gh auth login

# entrar na pasta do projeto
cd /caminho/do/seu/projeto

# iniciar git
git init

# criar .gitignore básico
cat > .gitignore << 'EOF'
.venv/
__pycache__/
*.pyc
.env
*.sqlite3
EOF

# adicionar arquivos
git add .

# primeiro commit
git commit -m "Initial commit"

# criar repositório público no GitHub e conectar remoto
gh repo create st-training-apps --public --source=. --remote=origin

# enviar para o GitHub
git branch -M main
git push -u origin main


----- dev rotine:

# ver arquivos alterados
git status

# adicionar todos os arquivos novos/alterados
git add .

# criar commit
git commit -m "Descreva o que foi alterado"

# enviar para o remoto
git push

-- one line
# fluxo completo em uma linha
git add . && git commit -m "Descreva o que foi alterado" && git push
